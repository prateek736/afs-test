function loadAFSScript() { 
 
    const adContainer = document.getElementById("afscontainer1"); 
    if (!adContainer) return; 
 
    if (window.afsLoaded) return; 
    window.afsLoaded = true; 
 
    (function(g,o){ 
        g[o]=g[o]||function(){(g[o]['q']=g[o]['q']||[]).push(arguments)}; 

--- PAGE 3 ---
        g[o]['t']=1*new Date; 
    })(window,'_googCsa'); 
 
    var script = document.createElement("script"); 
    script.src = "https://www.google.com/adsense/search/ads.js"; 
    script.async = true; 
 
    script.onload = function () { 
        runAFSAds(false); 
    }; 
 
    document.head.appendChild(script); 
} 
 
 
// 🧠 UNIVERSAL QUERY BUILDER 
function buildAFSQuery(isRetry = false) { 
 
    let category = document.getElementById("cat_name_ad")?.value?.trim() || ""; 
    let country = document.getElementById("countryName")?.value?.trim() || ""; 
    let audience = document.getElementById("event_audience")?.value || ""; 
    let format = document.getElementById("event_format")?.value || ""; 
 
    let isSearchPage = document.getElementById("is_search_page")?.value == "1"; 
    let searchPhrase = document.getElementById("search_phrase")?.value?.trim() || ""; 
 
    // 🟢 RETRY LOGIC (broad fallback) 
    if (isRetry) { 
        if (isSearchPage && searchPhrase) { 
            return searchPhrase; // retry without forcing "events" 
        } 
        return "business events"; 
    } 
 
    // 🟢 LISTING PAGE LOGIC 
    if (isSearchPage) { 
 
        let phrase = searchPhrase; 
 
        // fallback to H1 if no search phrase 
        if (!phrase) { 
            let h1 = document.querySelector("h1")?.innerText?.trim() || ""; 
            phrase = h1; 
        } 

--- PAGE 4 ---
 
        if (!phrase) return "business events"; 
 
        let lower = phrase.toLowerCase(); 
 
        // append "events" if missing 
        if (!/(^|\s)(event|events)(\s|$)/i.test(lower)) { 
            phrase += " events"; 
        } 
 
        return phrase; 
    } 
 
    // 🟡 DETAIL PAGE LOGIC 
 
    let noun = "business events"; 
 
    if (audience === "11000") { 
        if (format == "1") { 
            noun = "trade shows"; 
        } else if (format == "2") { 
            noun = "conferences"; 
        } 
    } else if (audience === "10100") { 
        noun = "events"; 
    } 
 
    let parts = []; 
 
    if (category) parts.push(category); 
    parts.push(noun); 
    if (country) parts.push(country); 
 
    return parts.join(" ").trim() || "business events"; 
} 
 
 
// 🚀 RUN ADS 
function runAFSAds(isRetry = false) { 
 
    let query = buildAFSQuery(isRetry); 
 
    var pageOptions = { 
        pubId: "partner-pub-8525015516580200", 

--- PAGE 5 ---
        query: query, 
        styleId: "4287184448", 
        adsafe: "medium" 
    }; 
 
    var adblock1 = { 
        container: "afscontainer1", 
        number: 2 
    }; 
 
    var rsblock1 = { 
        container: "relatedsearches1", 
        relatedSearches: 10 
    }; 
 
    _googCsa('ads', pageOptions, adblock1, rsblock1); 
 
    setTimeout(() => checkAFSFill(isRetry), 1800); 
} 
 
 
// 🔍 CHECK FILL + RETRY + FALLBACK 
function checkAFSFill(isRetry) { 
 
    const adContainer = document.getElementById("afscontainer1"); 
    const relatedContainer = document.getElementById("relatedsearches1"); 
    const fallback = document.getElementById("afs_fallback"); 
 
    if (!adContainer) return; 
 
    const hasAds = adContainer.querySelector("iframe"); 
 
    if (hasAds) { 
        if (relatedContainer) relatedContainer.style.display = "none"; 
        if (fallback) fallback.style.display = "none"; 
        return; 
    } 
 
    // 🔁 Retry once 
    if (!isRetry) { 
        console.log("AFS retry triggered"); 
        runAFSAds(true); 
        return; 
    } 

--- PAGE 6 ---
 
    // ❌ Final fallback 
    console.log("AFS fallback triggered"); 
 
    if (relatedContainer) relatedContainer.style.display = "block"; 
    if (fallback) fallback.style.display = "block"; 
 
    adContainer.style.display = "none"; 
}
